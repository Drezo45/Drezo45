function contigfileUpload(){
      var currenttime=new Date().valueOf();
      if (document.getElementById("contigUpload").value!=""){
        $.ajaxFileUpload({
          url:'searchc40.php',
          secureuri:false,
          fileElementId:'contigUpload',
          dataType: 'json',
          data:{ctime:currenttime},
          success: function (data, status)
          {
                if(typeof(data.error) != 'undefined')
                {
                        if(data.error != '')
                        {
                            alert(data.error);
                        }else{
                            document.getElementById("contiguploadtxt").innerHTML="uploadfile:"+currenttime+"  done!";
                            count_dddfromfiles();
                        }
                }
          },
          error: function (data, status, e)
          {
                alert(e);
          }
        });
        $.ajax({
         url: 'filepick.php',
         cache: false,
         dataType: 'html',
             type:'POST',
         data: {filename:currenttime},
         error: function(xhr) {
           alert('Ajax request error!');
         },
         success: function(msg){
            document.getElementById("contiguploadlist").innerHTML=msg;
         }
         });
      }else{
         if (document.getElementById("contigid").value!=""){
              $.ajax({
              url: 'searchc401.php',
              cache: false,
              dataType: 'html',
              type:'POST',
              data: {ctime:currenttime,id:$('#contigid').val()},
              beforeSend:function(){
                  showmask();
              },
              error: function(xhr) {
                  alert('Ajax request error!');
              },
              success: function(msg){
                  document.getElementById("contiguploadtxt").innerHTML="Save to file:"+currenttime+"  done!";
                  count_dddfromfiles();
              },
              complete:function(){
                  hidemask();
              }
            });
         }
      }
      return false;
}

function goto_contig(id) {
    xmlhttp=GetXmlHttpObject();
    if (xmlhttp==null) {
        alert ("Your browser does not support AJAX!");
        return;
    }
    var url="detail.php";
    url=url+"?id="+id;
    xmlhttp.onreadystatechange=function() {
        if (xmlhttp.readyState==4 || xmlhttp.readyState=="complete") {
            document.getElementById('tablecontent').innerHTML=xmlhttp.responseText;
        }
    }
    params="id="+id;
    xmlhttp.open("POST", url, true);
    xmlhttp.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
    xmlhttp.setRequestHeader("Content-length", params.length);
    xmlhttp.setRequestHeader("Connection", "close");
    xmlhttp.send(params);
}

function goto_contig(id) {
    xmlhttp=GetXmlHttpObject();
    if (xmlhttp==null) {
        alert ("Your browser does not support AJAX!");
        return;
    }
    var url="detail.php";
    url=url+"?id="+id;
    xmlhttp.onreadystatechange=function() {
        if (xmlhttp.readyState==4 || xmlhttp.readyState=="complete") {
            document.getElementById('tablecontent').innerHTML=xmlhttp.responseText;
        }
    }
    params="id="+id;
    xmlhttp.open("POST", url, true);
    xmlhttp.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
    xmlhttp.setRequestHeader("Content-length", params.length);
    xmlhttp.setRequestHeader("Connection", "close");
    xmlhttp.send(params);
}
function GetXmlHttpObject() {
    var xmlhttp=null;
    try {
        // Firefox, Opera 8.0+, Safari
        xmlhttp=new XMLHttpRequest();
    }
    catch (e) {
        // Internet Explorer
        try {
            xmlhttp=new ActiveXObject("Msxml2.XMLHTTP");
        }
        catch (e) {
            xmlhttp=new ActiveXObject("Microsoft.XMLHTTP");
        }
    }
    return xmlhttp;
}

function modify_pool(id){
   var target=-1;
   var table = document.getElementById("pool");

   for (var index=1;index<i;index++){
     if (table.rows[index].cells[3].innerHTML.indexOf(id)>0){
         target=index;
     }
   }
   if (target>=-1){
     var line="";
     var lined="";
     for (var index=1;index<i;index++){
       if (index!=target){
          data=table.rows[index].cells[1].innerHTML.split(">");
           if (data.length>1){
            line=data[1].toLowerCase().replace("</div","");
           }
        }else{
         data=table.rows[index].cells[1].innerHTML.split(">");
         if (data.length>1){
           lined=data[1].toLowerCase().replace("</div","");
          }
       }
     }
     var names=line.split(",");
     var mark=lined.split(",");
     $("input[name='c1[]' ]").each(function(){
            var value=$(this).val().toLowerCase();
            if( jQuery.inArray( value, names)>-1){
                $(this).attr('checked',false);
                $(this).attr('disabled',true);
            }else{
                if (jQuery.inArray( value, mark)>-1){
                  $(this).prop('disabled',false);
                  $(this).prop("checked",true);
                }else{
                $(this).attr('checked',false);
                $(this).attr('disabled',false);
                  }
            }
      });
   }
return false;
}

var objectIdToSlideDown = false;

function showHideContent(e,inputId)
{
        document.getElementById("Img").src="minus.gif";
        if(dhtmlgoodies_slideInProgress)return;
        dhtmlgoodies_slideInProgress = true;
        if(!inputId)inputId = this.id;
        inputId = inputId + '';
        var numericId = inputId.replace(/[^0-9]/g,'');
        var answerDiv = document.getElementById('aoption_a' + numericId);
        objectIdToSlideDown = false;
        if(!answerDiv.style.display || answerDiv.style.display=='none'){
                if(dhtmlgoodies_activeId &&  dhtmlgoodies_activeId!=numericId && !dhtmlgoodies_expandMultiple){
                        objectIdToSlideDown = numericId;
                        slideContent(dhtmlgoodies_activeId,(dhtmlgoodies_slideSpeed*-1));

                        answerDiv.style.display='block';
                        answerDiv.style.visibility = 'visible';
                        slideContent(numericId,dhtmlgoodies_slideSpeed);
                }
        }else{
                slideContent(numericId,(dhtmlgoodies_slideSpeed*-1));
                dhtmlgoodies_activeId = false;
                document.getElementById("Img").src="plus.gif";
        }
}
function slideContent(inputId,direction)
{
        var obj =document.getElementById('aoption_a' + inputId);
        var contentObj = document.getElementById('aoption_ac' + inputId);
        height = obj.clientHeight;
        if(height==0)height = obj.offsetHeight;
        height = height + direction;
        rerunFunction = true;
        if(height>contentObj.offsetHeight){
                height = contentObj.offsetHeight;
                rerunFunction = false;

                setTimeout('slideContent(' + inputId + ',' + direction + ')',dhtmlgoodies_timer);

                if(height<=1){

                                dhtmlgoodies_slideInProgress = false;
                        }

                        dhtmlgoodies_slideInProgress = false;
                }
        }
}

<p>&nbsp;&nbsp;&nbsp;&nbsp;Next generation sequencing (NGS) technologies bring the gene profiling study into big-data science era. However, the increasing amount of data made itself a problem for viewing data and deciphering the biological implication from it.</p>
<p>&nbsp;&nbsp;&nbsp;&nbsp;In this study,we provide a robust web platform to retrieve the gene (by its transcripts) expression data . Starting from columwise data of RNA-Seq mapping results (FPKM/RPKM), the build-in annotation database attached to the gene expression data; thus, biological meaning of the complex data is unlocked in the intuitive interface.</p>
<p>&nbsp;&nbsp;&nbsp;&nbsp;Currently, the platform accepts transcript expression data table in FPKM value, derived from Cufflinks or other tools that map reads to Forcipomyia taiwana genome. Please check [<a href="ftwissues.html" target="_blank">help/important issue</a>] section for the details about F. taiwana genome and General Feature Format(GFF).</p>
<p>&nbsp;&nbsp;&nbsp;&nbsp;Otherwise, you can also use the [<a href="https://hub.docker.com/r/lsbnb/docexpress_fastqc_ftw">docexpress_fastqc_ftw</a>] which is a docker image with built-in F. taiwana genome for generating the FPKM table through your own RNA-Seq data. [<a href="docexpress_fastqc_ftw.pdf">User manual for docexpress_fastqc_ftw </a>]</p>
